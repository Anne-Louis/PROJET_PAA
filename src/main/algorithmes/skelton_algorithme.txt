Input: 
    S = réseau initial
    epsilon = 0.5 //paramètre de régulation
Output: 
    minReseau = réseau optimisé

1. minReseau <- copie de S
2. meilleurCout <- coût(minReseau)

3. While epsilon > 0.01:
    3.1 courant <- copie de minReseau
    3.2 Supprimer toutes les connexions de courant
    3.3 Trier les maisons de courant par consommation décroissante
    3.4 maisonsNonAllouees <- []

    3.5 For chaque maison m dans courant:
        3.5.1 Calculer moyenneTaux des générateurs
        3.5.2 Trier générateurs par capacité restante décroissante
        3.5.3 alloue <- faux

        3.5.4 For chaque générateur g:
            If (capacité restante de g >= consommation de m) AND (|taux(g) - moyenneTaux| <= epsilon):
                Allouer m à g
                alloue <- vrai
                Break

        3.5.5 If alloue = faux:
            Ajouter m à maisonsNonAllouees

    3.6 For chaque maison m dans maisonsNonAllouees:
        Tester toutes allocations possibles
        Choisir celle minimisant le coût
        Mettre à jour courant

    3.7 If coût(courant) < meilleurCout:
        minReseau <- courant
        meilleurCout <- coût(courant)

    3.8 epsilon <- epsilon * 0.99

4. Return minReseau
